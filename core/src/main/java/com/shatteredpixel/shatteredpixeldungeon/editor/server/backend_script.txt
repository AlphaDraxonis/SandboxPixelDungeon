function doPost(e) {

  if (e.parameter.action == ACTION_UPLOAD_START) {

    console.log(ACTION_UPLOAD_START);

    const user = e.parameter.userID;
    const userIDFileContent = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };

    if (isUserBanned(userIDFileContent.id1, userIDFileContent.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const uploadTempID = createDirectory(new Date().getTime() + Math.random(), UPLOAD_TEMP);

    const dungeonID = createDirectory(e.parameter.title, uploadTempID)

    //Hier ist aber der echte Ordner, der dann auch tatsächlich verschoben wird. Deshalb hat er auch bereits den richtigen Namen. Seine ID ist besonders wichtig.
    const versionID = createDirectory("1", dungeonID);//we start uploading, so it will always be version 1

    //hier wird der Autor im TempDir registriert und mit der entsprechenden DungeonID assoziiert.
    uploadFile(JSON.stringify(userIDFileContent), dungeonID, "", uploadTempID, false);


    const result = {
      dungeonID: dungeonID,
      versionID: versionID
    }
    return ContentService.createTextOutput(KEYWORD_SUCCESS + JSON.stringify(result))
      .setMimeType(ContentService.MimeType.TEXT);
  }
  else if (e.parameter.action == ACTION_UPLOAD_FINISH) {

    console.log(ACTION_UPLOAD_FINISH);

    const user = e.parameter.userID;
    const userIDs = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };

    if (isUserBanned(userIDs.id1, userIDs.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const dungeonID = e.parameter.dungeonID;
    const versionID = e.parameter.versionID;

    const dungeonFolder = DriveApp.getFolderById(dungeonID);

    //verify that the folder given by versionID actually belongs to the dungeon
    if (!versionBelongsToDungeon(versionID, dungeonID)) {
      return ContentService.createTextOutput(KEYWORD_NOT_THE_CREATOR)//actually the dungeon id was not found but we don’t want to give away this information
      .setMimeType(ContentService.MimeType.TEXT);
    }


    let parents = dungeonFolder.getParents();

    parents = dungeonFolder.getParents();
    let dungeonTemp;
    if (parents.hasNext()) dungeonTemp = parents.next();

    if (!isCreatorDuringUpload(dungeonTemp, dungeonID, userIDs.id1, userIDs.id2)) return ContentService.createTextOutput(KEYWORD_NOT_THE_CREATOR)
      .setMimeType(ContentService.MimeType.TEXT);

    movePreviewImagesFromVersionDirToDungeonDir(e.parameter.versionID, dungeonID);

    //Nun muss die Versionsdatei erstellt werden:
    const versionInfo = [];
    versionInfo.push(createNewVersionEntry(e));
    const versionFileId = uploadFile(JSON.stringify(versionInfo), FILE_NAME_VERSIONS, "", dungeonFolder.getId(), false);

    //Erstellt die PreviewFile (zunächst im TempDir), damit der Dungeon überhaupt angezeigt werden kann.
    const previewFileID = uploadPreviewFile(e, userIDs, dungeonTemp.getId(), versionFileId, "Uploaded");

    //Verschiebt den ins TempDir hochgeladenen Dungeon inkl. der darin befindlichen ersten Version ins Full-Dungeon-Directory
    dungeonFolder.moveTo(DriveApp.getFolderById(FULL_DUNGEON_FOLDER));

    //Hier wird jetzt der Autor richtig registriert
    dungeonTemp.getFilesByName(dungeonID).next().moveTo(DriveApp.getFolderById(USER_ID_FOLDER));
    //und zum Schluss die PreviewFile...
    DriveApp.getFileById(previewFileID).moveTo(DriveApp.getFolderById(PREVIEW_FOLDER));

    //entfernt das Temporary Directory
    dungeonTemp.setTrashed(true);

    return ContentService.createTextOutput(KEYWORD_SUCCESS + dungeonID)
      .setMimeType(ContentService.MimeType.TEXT);

  }
  else if (e.parameter.action === ACTION_UPLOAD_FILE) {

    console.log(ACTION_UPLOAD_FILE);

    const user = e.parameter.userID;
    const userIDs = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };

    if (isUserBanned(userIDs.id1, userIDs.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const dungeonID = e.parameter.dungeonID;
    const versionID = e.parameter.versionID;

    const versionFolder = DriveApp.getFolderById(versionID);


    let isCreator = false;

    let parents = versionFolder.getParents();
    let dungeonFolder;
    if (parents.hasNext()) dungeonFolder = parents.next();

    isCreator = isCreatorDuringUpload(dungeonFolder, dungeonID, userIDs.id1, userIDs.id2);

    parents = dungeonFolder.getParents();
    let dungeonTemp;
    if (parents.hasNext()) dungeonTemp = parents.next();

    if (!isCreator && !isCreatorDuringUpload(dungeonTemp, dungeonID, userIDs.id1, userIDs.id2)) return ContentService.createTextOutput(KEYWORD_NOT_THE_CREATOR)
      .setMimeType(ContentService.MimeType.TEXT);


    const fileName = e.parameter.fileName;

    uploadFile(e.parameter.content, fileName, "", versionID, true);

    return ContentService.createTextOutput(KEYWORD_SUCCESS + versionID)
      .setMimeType(ContentService.MimeType.TEXT);
  }

  else if (e.parameter.action === ACTION_UPDATE_START) {

    console.log(ACTION_UPDATE_START);

    const dungeonID = e.parameter.dungeonID;

    if (!isValidFolder(dungeonID)) {
      return ContentService.createTextOutput(KEYWORD_NOT_THE_CREATOR)//actually the dungeon ID was not found but we don’t want to give away this information
      .setMimeType(ContentService.MimeType.TEXT);
    }

    const user = e.parameter.userID;
    const userIDFileContent = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };

    if (isUserBanned(userIDFileContent.id1, userIDFileContent.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    if (!isCreator(dungeonID, userIDFileContent.id1, userIDFileContent.id2)) return ContentService.createTextOutput('UserID is incorrect!')
      .setMimeType(ContentService.MimeType.TEXT);

    const dungeonFolder = DriveApp.getFolderById(dungeonID);

    //erstellt TempDir, in den die einzelnen Dateinen dieser Dungeonversion beim Upload geladen werden. (Später werden sie dann an den richtigen Platz verschoben)
    const uploadTempID = createDirectory(new Date().getTime() + Math.random(), UPLOAD_TEMP);

    let versionID = "";

    const includesNewVersion = e.parameter.includesNewVersion === "true";

    if (includesNewVersion) {
      //we need to create a new dir for the new version
      //TODO find the correct version name
      //dies durchsuche dazu den Ordner zu dungeonID nach Ordnern, deren Name ein Integer ist. Der höchste Name plus 1 ist dann der neue Name

      //finde einen Namen für die nächste Version
      //der Name ist einfach die vorherige höchste Versionsnummer + 1
      let maxVersion = 0;
      const subfolders = dungeonFolder.getFolders();
      while (subfolders.hasNext()) {
        const folder = subfolders.next();

        const versionNumber = parseInt(folder.getName(), 10);
        if (!isNaN(versionNumber)) {
          maxVersion = Math.max(maxVersion, versionNumber);
        }
      }

      const nextVersionName = maxVersion + 1;

      versionID = createDirectory(nextVersionName.toString(), uploadTempID);
    }

    //hier wird der Autor im TempDir registriert und mit der entsprechenden DungeonID assoziiert.
    uploadFile(JSON.stringify(userIDFileContent), dungeonID, "", uploadTempID, false);

    const result = {
      dungeonID: dungeonID,
      versionID: versionID,
      uploadTempID: uploadTempID
    }
    return ContentService.createTextOutput(KEYWORD_SUCCESS + JSON.stringify(result))
      .setMimeType(ContentService.MimeType.TEXT);

  }

  else if (e.parameter.action === ACTION_UPDATE_FINISH) {

    console.log(ACTION_UPDATE_FINISH);

    const user = e.parameter.userID;
    const userIDs = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };

    if (isUserBanned(userIDs.id1, userIDs.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const includesNewVersion = (e.parameter.includesNewVersion === 'true');

    const dungeonID = e.parameter.dungeonID;
    const versionID = e.parameter.versionID;
    const uploadTempID = e.parameter.uploadTempID;

    const uploadTemp = DriveApp.getFolderById(uploadTempID);

    //verify that the folder given by versionID actually belongs to the dungeon, if one is provided
    if (includesNewVersion && !versionBelongsToDungeon(versionID, uploadTempID)) {
      return ContentService.createTextOutput("Illegal arguments: incorrect version folder id given.")
      .setMimeType(ContentService.MimeType.TEXT);
    }

    if (!isCreatorDuringUpload(uploadTemp, dungeonID, userIDs.id1, userIDs.id2)) return ContentService.createTextOutput(KEYWORD_NOT_THE_CREATOR)
      .setMimeType(ContentService.MimeType.TEXT);


    const existingPreviewFile = readPreviewFile(dungeonID);

    if (!includesNewVersion) {
      e.parameter.int_version = existingPreviewFile.int_version;
      e.parameter.version = existingPreviewFile.version;

      //we don’t upload a new dungeon version, so the old one is still the newest
      e.parameter.versionID = existingPreviewFile.most_recent_dungeon;
    }

    let newVersionFileID = existingPreviewFile.version_file_id;

    if (includesNewVersion) {
      //Nun muss die Versionsdatei aktualisiert werden:
      const versionInfo = readVersionInfo(existingPreviewFile.version_file_id);
      versionInfo.push(createNewVersionEntry(e));
      newVersionFileID = uploadFile(JSON.stringify(versionInfo), FILE_NAME_VERSIONS, "", uploadTempID, false);
    }

    if (includesNewVersion) {
      //die Bilder müssen vor der Erstellung der PreviewFile aktualisiert werden
      movePreviewImagesFromVersionDirToDungeonDir(versionID, dungeonID);
    }

    //Erstellt nun die PreviewFile (zunächst im TempDir)
    const previewFileID = uploadPreviewFile(e, userIDs, uploadTempID, newVersionFileID, "Updated");

    const dungeonFolder = DriveApp.getFolderById(dungeonID);

    if (includesNewVersion) {
      //Verschiebt die ins TempDir hochgeladene Version in den zugehörigen Dungeon-Ordner im Full-Dungeon-Directory
      const versionFolder = DriveApp.getFolderById(versionID);
      versionFolder.moveTo(dungeonFolder);
    }

    //Jetzt wird die VersionInfoFile ausgetauscht
    DriveApp.getFileById(existingPreviewFile.version_file_id).setTrashed(true);
    DriveApp.getFileById(newVersionFileID).moveTo(DriveApp.getFolderById(dungeonFolder.getId()));

    //Zum Schluss wird die PreviewFile ausgetauscht
    //dazu wird zunächst die bisherige PreviewFile gelöscht
    const allPreviewFiles = DriveApp.getFolderById(PREVIEW_FOLDER).getFiles();
    while (allPreviewFiles.hasNext()) {
      const file = allPreviewFiles.next();
      if (file.getName().endsWith(dungeonID)) {
        file.setTrashed(true);
      }
    }
    //und jetzt die Datei verschoben
    DriveApp.getFileById(previewFileID).moveTo(DriveApp.getFolderById(PREVIEW_FOLDER));


    //entfernt das Temporary Directory (inkl. der temporären Autor-Verifikation)
    uploadTemp.setTrashed(true);

    return ContentService.createTextOutput(KEYWORD_SUCCESS + dungeonID)
      .setMimeType(ContentService.MimeType.TEXT);

  }

  else if (e.parameter.action === ACTION_BUG_REPORT) {

    console.log(ACTION_BUG_REPORT);

    const user = e.parameter.userID;

    if (isUserBanned(hashUserIDOne(user), hashUserIDTwo(user))) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const fileName = e.parameter.fileName;

    let dungeon = e.parameter.dungeon;

    let dungeonID = "none"
    if (dungeon !== "null") {
      dungeonID = uploadFile(dungeon, fileName, ".dun", BUG_REPORT_FOLDER, true);
    }

    const preview = {
      title: e.parameter.title,
      description: e.parameter.description,
      version: e.parameter.version,
      dungeonFile: dungeonID,
      upload_time: new Date().getTime(),
    };

    sendNotificationEmail(fileName, "Bug report", e.parameter.description, e.parameter.version);

    return ContentService.createTextOutput(KEYWORD_SUCCESS)
      .setMimeType(ContentService.MimeType.TEXT);

  }
  else if (e.parameter.action === ACTION_CANCEL) {

    console.log(ACTION_CANCEL);

    //TODO, insbesondere KANN das gar nicht funktionieren, während man aktualisiert!

    const user = e.parameter.userID;
    const userIDs = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };

    if (isUserBanned(userIDs.id1, userIDs.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const dungeonID = e.parameter.dungeonID;
    const userIdFileName = e.parameter.userIdFileName;

    const dungeonFolder = DriveApp.getFolderById(dungeonID);
    const parents = dungeonFolder.getParents();

    let uploadTemp;
    if (parents.hasNext()) uploadTemp = parents.next();

    if (!isCreatorDuringUpload(uploadTemp, userIdFileName, userIDs.id1, userIDs.id2)) {
      return ContentService.createTextOutput(KEYWORD_NOT_THE_CREATOR)
        .setMimeType(ContentService.MimeType.TEXT);
    }

    uploadTemp.setTrashed(true);

    return ContentService.createTextOutput(KEYWORD_SUCCESS + dungeonID)
      .setMimeType(ContentService.MimeType.TEXT);

  }
  else {
    createAutomaticErrorReport("Invalid POST action", "Invalid POST action: " + e.parameter.action);
    return ContentService.createTextOutput("Invalid POST action: " + e.parameter.action);
  }
}


function doGet(e) {
  if (e.parameter.action === ACTION_GET_PREVIEW_LIST) {
    const result = listFiles(PREVIEW_FOLDER, e.parameter.page, e.parameter.perPage);
    return ContentService.createTextOutput(result)
      .setMimeType(ContentService.MimeType.JSON);
  }

  else if (e.parameter.action === ACTION_GET_LATEST_UPLOAD_TIME) {

    return ContentService.createTextOutput(getLatestUploadTime().toString())
        .setMimeType(ContentService.MimeType.TEXT);
  }

  else if (e.parameter.action === ACTION_START_DOWNLOAD) {

    const userID = e.parameter.userID;
    const folderID = e.parameter.folderID;

    const folder = DriveApp.getFolderById(folderID);

    const fileIDs = getAllFileIdsInFolder(folder);

    fileIDs.unshift({creator: isCreator(folderID, hashUserIDOne(userID), hashUserIDTwo(userID))});

    return ContentService.createTextOutput(JSON.stringify(fileIDs))
      .setMimeType(ContentService.MimeType.JSON);
  }
  else if (e.parameter.action === ACTION_DOWNLOAD_FILE) {

    const content = getFileContent(e.parameter.fileID);
    return ContentService.createTextOutput(content)
      .setMimeType(ContentService.MimeType.TEXT);

  }
  else if (e.parameter.action === ACTION_READ_FILE) {

    const content = getFileContentUncompressed(e.parameter.fileID);
    return ContentService.createTextOutput(JSON.stringify(content))
      .setMimeType(ContentService.MimeType.JSON);

  }
  else if (e.parameter.action === ACTION_IS_CREATOR) {
    const userID = e.parameter.userID;

    const isCorrect = isCreator(e.parameter.dungeonID, hashUserIDOne(userID), hashUserIDTwo(userID));
    return ContentService.createTextOutput(isCorrect.toString())
      .setMimeType(ContentService.MimeType.TEXT);
  }
  else if (e.parameter.action === ACTION_DELETE_DUNGEON) {

    const user = e.parameter.userID;
    const userIDFileContent = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };
    if (isUserBanned(userIDFileContent.id1, userIDFileContent.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const result = deleteDungeon(e.parameter.dungeonID, userIDFileContent.id1, userIDFileContent.id2);
    return ContentService.createTextOutput(result.toString())
      .setMimeType(ContentService.MimeType.TEXT);
  }
  else if (e.parameter.action === ACTION_DELETE_VERSION) {

    const user = e.parameter.userID;
    const userIDFileContent = {
      id1: hashUserIDOne(user),
      id2: hashUserIDTwo(user),
    };
    if (isUserBanned(userIDFileContent.id1, userIDFileContent.id2)) return ContentService.createTextOutput(KEYWORD_BANNED)
      .setMimeType(ContentService.MimeType.TEXT);

    const content = deleteVersion(e.parameter.dungeonID, e.parameter.versionID, userIDFileContent.id1, userIDFileContent.id2);
    return ContentService.createTextOutput(KEYWORD_SUCCESS + JSON.stringify(content))
      .setMimeType(ContentService.MimeType.TEXT);
  }
  else {
    createAutomaticErrorReport("Invalid GET action", "Invalid GET action: " + e.parameter.action);
    return ContentService.createTextOutput("Invalid GET action: " + e.parameter.action);
  }
}



function createNewVersionEntry(e) {
    return {
      version_name: e.parameter.versionName,
      upload_time: new Date().getTime(),
      version_id: e.parameter.versionID
    };
}

function uploadPreviewFile(e, userIDs, uploadTempID, versionFileId, action) {
  const preview = {
      title: e.parameter.title,
      description: e.parameter.description,
      dungeonID: e.parameter.dungeonID,
      most_recent_dungeon: e.parameter.versionID,
      version: e.parameter.version,
      int_version: e.parameter.int_version,
      uploader: e.parameter.uploader,
      difficulty: e.parameter.difficulty,
      upload_time: new Date().getTime(),
      version_file_id: versionFileId,
      is_debug: false,
      userHash: userIDs.id1.substring(0, 10),
      preview_image_file_ids: getPreviewImageFileIds(e.parameter.dungeonID),
    };
    const fileID = uploadFile(JSON.stringify(preview), preview.upload_time + "_" + e.parameter.dungeonID, "", uploadTempID, false);

    if (!preview.is_debug)
      sendNotificationEmail(e.parameter.title, action, e.parameter.description, e.parameter.int_version);

    return fileID;
}

function getPreviewImageFileIds(folderId) {
  const folder = DriveApp.getFolderById(folderId);
  const files = folder.getFiles();
  var previewFileIds = [];

  while (files.hasNext()) {
    const file = files.next();
    const fileName = file.getName();

    if (file.getMimeType() === MimeType.FOLDER) {
      continue;
    }

    const match = fileName.match(/^preview(\d+)\.png$/);
    if (match) {
      var fileNumber = parseInt(match[1], 10);
      previewFileIds.push({ id: file.getId(), number: fileNumber });
    }
  }

  previewFileIds.sort(function(a, b) {
    return a.number - b.number;
  });

  return previewFileIds.map(function(file) {
    return file.id;
  });
}

function movePreviewImagesFromVersionDirToDungeonDir(versionID, dungeonID) {

  const versionFolder = DriveApp.getFolderById(versionID);
  const dungeonFolder = DriveApp.getFolderById(dungeonID);

  /*//make sure this operation can only be applied where we want it
  var parents = dungeonFolder.getParents();
  if (!parents.hasNext() || parents.next().getId() !== FULL_DUNGEON_FOLDER) {
    createAutomaticErrorReport("Security check failed", "Dungeon folder is not a child of FULL_DUNGEON_FOLDER.");
    throw new Error('Security check failed: Dungeon folder is not a child of FULL_DUNGEON_FOLDER.');
  }*/

  //delete all pre-existing previewImage files
  const dungeonFiles = dungeonFolder.getFiles();
  while (dungeonFiles.hasNext()) {
    const file = dungeonFiles.next();
    const fileName = file.getName();

    if (file.getMimeType() === MimeType.FOLDER) {
      continue; //Skip folders
    }

    if (fileName.match(/^preview(\d+)\.png$/)) {
      file.setTrashed(true);
    }

  }

  //move the new previewImages to the dungeon folder
  const versionFiles = versionFolder.getFiles();
  while (versionFiles.hasNext()) {
    const file = versionFiles.next();
    const fileName = file.getName();

    if (file.getMimeType() === MimeType.FOLDER) {
      continue; //Skip folders
    }

    if (fileName.match(/^preview(\d+)\.png$/)) {
      file.moveTo(dungeonFolder);
    }

  }
}

function getLatestUploadTime() {
  const folder = DriveApp.getFolderById(FULL_DUNGEON_FOLDER);

  let latestModifiedDate = 0;

  const files = folder.getFolders();
  while (files.hasNext()) {
    var fileModifiedDate = files.next().getLastUpdated().getTime();

    if (fileModifiedDate > latestModifiedDate) {
      latestModifiedDate = fileModifiedDate;
    }
  }

  return latestModifiedDate;
}

function createDirectory(name, parentDirectoryID) {
  const folder = DriveApp.getFolderById(parentDirectoryID);
  return folder.createFolder(name).getId();
}

function uploadFile(content, fileName, fileExtension, parentFolderID, compress) {

  let folders = fileName.split('/');
  console.log(parentFolderID);
  let parentFolder = DriveApp.getFolderById(parentFolderID);

  let lock = LockService.getScriptLock();
  lock.waitLock(300000); // Wait up to 5 min for a lock to free up if it doesn't open itself, throw an error in that case

  try {
    for (let i = 0; i < folders.length - 1; i++) {
      let folderName = folders[i];
      let subFolder;

      let subFolders = parentFolder.getFoldersByName(folderName);
      if (subFolders.hasNext()) {
        subFolder = subFolders.next();
      } else {
        subFolder = parentFolder.createFolder(folderName);
      }
      parentFolder = subFolder;
    }

    let finalFileName = folders[folders.length - 1];


    if (fileExistsInFolder(parentFolder, finalFileName + fileExtension)) {
      let count = 1;
      let originalFileName = finalFileName;
      do {
        finalFileName = originalFileName + count;
        count++;
      } while (fileExistsInFolder(parentFolder, finalFileName + fileExtension));
    }

    let blob;
    if (compress) {
      blob = Utilities.newBlob(content);
      const fileBytes = blob.getBytes();
      blob = compressGZip(fileBytes);
      blob.setName(finalFileName + fileExtension);
    } else {
      blob = Utilities.newBlob(content, 'application/octet-stream', finalFileName + fileExtension);
    }

    return parentFolder.createFile(blob).getId();
  } finally {
    lock.releaseLock();
  }
}

function deleteDungeon(dungeonID, id1, id2) {

  if (!isCreator(dungeonID, id1, id2)) return 'UserID is incorrect!';

  DriveApp.getFileById(dungeonID).setTrashed(true);

  const allPreviewFiles = DriveApp.getFolderById(PREVIEW_FOLDER).getFiles();
  while (allPreviewFiles.hasNext()) {
    const file = allPreviewFiles.next();
    if (file.getName().endsWith(dungeonID)) {
      file.setTrashed(true);
    }
  }

  const userIDFile = DriveApp.getFolderById(USER_ID_FOLDER).getFilesByName(dungeonID);

function deleteVersion(dungeonID, versionID, id1, id2) {

      return {

      };
    }
    //finde die jetzt neueste Version
    let latestVersion = versionInfo.reduce((latest, current) => {
      return new Date(current.upload_time) > new Date(latest.upload_time) ? current : latest;
    });
    e.parameter.versionID = latestVersion.version_id;

    e.paramater.versionID = existingPreviewFile.most_recent_dungeon;

  const updatedVersionInfo = versionInfo.filter(info => info.version_id !== versionID);

  //tausche die VersionInfoFile aus
  DriveApp.getFileById(existingPreviewFile.version_file_id).setTrashed(true);
  const newVersionFileID = uploadFile(JSON.stringify(updatedVersionInfo), FILE_NAME_VERSIONS, "", dungeonFolder.getId(), false);

  //Jetzt wird die PreviewFile ausgetauscht
  //dazu wird zunächst die bisherige PreviewFile gelöscht
  const allPreviewFiles = DriveApp.getFolderById(PREVIEW_FOLDER).getFiles();
  while (allPreviewFiles.hasNext()) {
    const file = allPreviewFiles.next();

  //...und nun durch eine Neue ersetzt
  const previewFileID = uploadPreviewFile(e, userIDs, PREVIEW_FOLDER, newVersionFileID, "Version deleted");

function listFiles(folderID, page, perPage) {

  const folder = DriveApp.getFolderById(folderID);
  const files = folder.getFiles();
  const fileList = [];

  while (files.hasNext()) {
    const file = files.next();
    fileList.push({
      id: file.getId(),
      name: file.getName(),
      dungeonID: file.getName().split('_').slice(1).join('_'),
    });
  }

  const numPreviews = fileList.length;

  fileList.sort((a, b) => b.name.localeCompare(a.name));

  const startIndex = page * perPage;
  const subset = fileList.slice(startIndex, startIndex + parseInt(perPage));

  subset.forEach(fileInfo => {
    const file = DriveApp.getFileById(fileInfo.id);
    fileInfo.content = file.getBlob().getDataAsString();
  });

  const numPreviewsObj = {
    numPreviews: numPreviews
  }
  subset.unshift(numPreviewsObj);

  return JSON.stringify(subset);
}


function fileExistsInFolder(folder, filename) {
  const files = folder.getFilesByName(filename);
  return files.hasNext();
}


function getAllFileIdsInFolder(folder, currentPath = '') {

  var result = [];

  var files = folder.getFiles();
  while (files.hasNext()) {
    var file = files.next();
    var filePath = currentPath + file.getName();
    if (!filePath.match(/^preview(\d+)\.png$/)) {
      result.push({ id: file.getId(), path: filePath});
    }
  }

  var subfolders = folder.getFolders();
  while (subfolders.hasNext()) {
    var subfolder = subfolders.next();
    var path = currentPath + subfolder.getName() + '/';
    result = result.concat(getAllFileIdsInFolder(subfolder, path));
  }

  return result;
}

function decompressGZip(compressedBlob) {

    return "Error during decompression:" + error;
  }
}

function compressGZip(decompressedBlob) {
  try {
    return Utilities.gzip(Utilities.newBlob(decompressedBlob));
  } catch (error) {
    console.error('Error during compression:', error);
    return "Error during compression:" + error;
  }
}


function isFileInFolderTree(fileId) {
  let currentFileId = fileId;

  while (true) {
    const file = DriveApp.getFileById(currentFileId);
    console.log(file.getName())

    const parents = file.getParents();

    if (parents === null || !parents.hasNext()) {
      return false;
    }

    const parentId = parents.next().getId();

    if (parentId === ROOT_FOLDER) {
      return true;
    }

    // Go up one level
    currentFileId = parentId;
  }
}


function versionBelongsToDungeon(versionID, dungeonID) {
  const versionFolder = DriveApp.getFolderById(versionID);
  const parents = versionFolder.getParents();
  while (parents.hasNext()) {

      return true;
    }
  }
  return false;
}

function hashUserID(userID, salt) {
  try {
    const combined = userID + salt;
    const hash = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, combined, Utilities.Charset.UTF_8);
    const hashedUserID = Utilities.base64Encode(hash);
    return hashedUserID;
  } catch (error) {
    console.error('Error hashing userID:', error);
    return null;
  }
}
